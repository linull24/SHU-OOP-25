\section{其他可变参数技术}

\begin{frame}[fragile]{\texttt{std::initializer\_list}}
  \begin{block}{同类型场景的解决方案}
    适用于所有参数类型相同的情况
  \end{block}
  
  \begin{lstlisting}[language=C++,basicstyle=\small\ttfamily]
void print(std::initializer_list<int> list) {
    for (int x : list) 
        std::cout << x << " ";
}

print({1, 2, 3});  
// 内部是 const T[]
  \end{lstlisting}
  
  \vspace{1em}
  \begin{itemize}
    \item 内部实现为 \texttt{const T[]}
    \item 只能用于相同类型的参数
    \item 语法简洁，易于使用
  \end{itemize}
\end{frame}
