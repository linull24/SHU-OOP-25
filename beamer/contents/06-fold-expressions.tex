\begin{frame}[fragile]{总结}
  \begin{itemize}
    \item C风格：\texttt{va\_list}, \texttt{va\_start}, \texttt{va\_arg}, \texttt{va\_end}
    \item 模板可变参数：\texttt{typename...}, \texttt{Args\&\&...}
    \item 完美转发：\texttt{std::forward}, \texttt{std::move}
    \item 类型约束：\texttt{static\_assert}, \texttt{concept}
    \item 同类型场景：\texttt{std::initializer\_list}
    \item 折叠表达式：C++17引入，简化参数包处理
  \end{itemize}
\end{frame}
